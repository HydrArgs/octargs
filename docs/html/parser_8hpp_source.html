185  bool parse_exclusive_recursively(const results_data_ptr_type& results_data_ptr, values_storage_type& values_storage,
186  argument_table_iterator& input_iterator) const
187  {
188  if (input_iterator.get_remaining_count() == 0)
189  {
190  // no arguments - exit
191  return false;
192  }
193  else if (input_iterator.get_remaining_count() == 1)
194  {
195  auto& arg_name = input_iterator.take_next();
196 
197  auto arg_iter = m_data_ptr->m_argument_repository.m_names_repository.find(arg_name);
198  if (arg_iter == m_data_ptr->m_argument_repository.m_names_repository.end())
199  {
200  // not an argument name
201  return false;
202  }
203 
204  auto& arg_object_ptr = arg_iter->second;
205 
206  if (!arg_object_ptr->is_exclusive())
207  {
208  // not exclusive
209  return false;
210  }
211 
212  auto& value_str = dictionary_type::get_switch_enabled_literal();
213 
214  parse_argument_value(results_data_ptr, values_storage, arg_object_ptr, arg_name, value_str);
215 
216  return true;
217  }
218  else
219  {
220  if (!m_data_ptr->m_argument_repository.m_subparsers_argument)
221  {
222  // no subparsers
223  return false;
224  }
225 
226  auto& arg_name = input_iterator.take_next();
227 
228  auto arg_iter = m_data_ptr->m_argument_repository.m_names_repository.find(arg_name);
229  if (arg_iter != m_data_ptr->m_argument_repository.m_names_repository.end())
230  {
231  // argument name, subparser expected
232  return false;
233  }
234 
235  auto subparser = m_data_ptr->m_argument_repository.m_subparsers_argument->get_parser(arg_name);
236  if (!subparser)
237  {
238  // not a subparser name
239  return false;
240  }
241 
242  return subparser->parse_exclusive_recursively(results_data_ptr, values_storage, input_iterator);
243  }

246  results_type parse_regular(const results_data_ptr_type& results_data_ptr, values_storage_type& values_storage,
247  argument_table_iterator& input_iterator) const
248  {
249  parse_named_arguments(values_storage, results_data_ptr, input_iterator);
250  if (m_data_ptr->m_argument_repository.m_subparsers_argument)
251  {
252  /* all remaining arguments will go to subparser so process this parser defaults & requirements */
253  parse_default_values(results_data_ptr, values_storage);
254  check_values_count(results_data_ptr);
255 
256  return parse_subparsers_argument(values_storage, results_data_ptr, input_iterator);
257  }
258  else
259  {
260  parse_positional_arguments(values_storage, results_data_ptr, input_iterator);
261 
262  if (input_iterator.has_more())
263  {
264  throw parser_error_ex<char_type>(
265  parser_error_code::SYNTAX_ERROR, string_type(), input_iterator.peek_next());
266  }
267 
268  parse_default_values(results_data_ptr, values_storage);
269  check_values_count(results_data_ptr);
270 
271  return results_type(results_data_ptr);
272  }

275  static void parse_argument_value(const results_data_ptr_type& results_data_ptr, values_storage_type& values_storage,
276  const const_argument_ptr_type& argument, const string_type& arg_name, const string_type& value_str)
277  {
278  auto count = results_data_ptr->value_count(argument);
279  if (count >= argument->get_max_count())
280  {
281  throw parser_error_ex<char_type>(parser_error_code::TOO_MANY_OCCURRENCES, arg_name, value_str);
282  }
283 
284  // check if the value is among the allowed
285  auto& allowed_values = argument->get_allowed_values();
286  if (!allowed_values.empty())
287  {
288  if (std::find(allowed_values.begin(), allowed_values.end(), value_str) == allowed_values.end())
289  {
290  throw parser_error_ex<char_type>(parser_error_code::VALUE_NOT_ALLOWED, arg_name, value_str);
291  }
292  }
293 
294  // if there is a handler call it first, to make sure the value
295  // is only stored if handler accepted it.
296  auto handler = argument->get_handler();
297  if (handler)
298  {
299  try
300  {
301  handler->parse(values_storage, value_str);
302  }
303  catch (const conversion_error&)
304  {
305  std::throw_with_nested(
306  parser_error_ex<char_type>(parser_error_code::CONVERSION_FAILED, arg_name, value_str));
307  }
308  }
309 
310  results_data_ptr->append_value(argument, value_str);
311  }

314  const const_argument_ptr_type& argument)
315  {

318  return;
319  }
320 
321  auto& values = argument->get_default_values();
322  if (values.empty())
323  {
324  return;
325  }
326 
327  for (auto& value : values)

342  bool parse_named_argument(const results_data_ptr_type& results_data_ptr, values_storage_type& values_storage,
343  argument_table_iterator& input_iterator, const string_type& arg_name) const
344  {
345  auto arg_iter = m_data_ptr->m_argument_repository.m_names_repository.find(arg_name);
346  if (arg_iter == m_data_ptr->m_argument_repository.m_names_repository.end())
347  {
348  // not an argument name, goto positional arguments processing
349  return false;
350  }
351 
352  auto& arg_object_ptr = arg_iter->second;
353 
354  if (!arg_object_ptr->is_assignable_by_name() || arg_object_ptr->is_exclusive())
355  {
356  // not an named argument, goto positional arguments processing
357  return false;
358  }
359 
360  // argument found, so remove element from input
361  input_iterator.take_next();
362 
363  string_type value_str;
364 
365  if (arg_object_ptr->is_accepting_separate_value())
366  {
367  if (!input_iterator.has_more())
368  {
369  throw parser_error_ex<char_type>(parser_error_code::VALUE_MISSING, arg_name, string_type());
370  }
371 
372  value_str = input_iterator.take_next();
373  }
374  else
375  {
376  value_str = dictionary_type::get_switch_enabled_literal();
377  }
378 
379  parse_argument_value(results_data_ptr, values_storage, arg_object_ptr, arg_name, value_str);
380 
381  return true;

413  bool parse_named_argument(const results_data_ptr_type& results_data_ptr, values_storage_type& values_storage,
414  argument_table_iterator& input_iterator) const
415  {
416  auto& input_value = input_iterator.peek_next();
417 
418  auto equal_char_pos = input_value.find(dictionary_type::get_equal_literal());
419  if (equal_char_pos == string_type::npos)
420  {
421  return parse_named_argument(results_data_ptr, values_storage, input_iterator, input_value);
422  }
423  else
424  {
425  auto name_str = input_value.substr(0, equal_char_pos);
426  auto value_str = input_value.substr(equal_char_pos + 1);
427 
428  return parse_named_argument(results_data_ptr, values_storage, input_iterator, name_str, value_str);
429  }
